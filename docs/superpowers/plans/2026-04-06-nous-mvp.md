# Nous MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a macOS-native personal knowledge management + AI assistant with local vector storage, knowledge graph, notes editor, and configurable LLM.

**Architecture:** All content (conversations + notes) are NousNodes stored in SQLite. Vectors stored as BLOBs with brute-force cosine similarity via Accelerate (fast enough for <10k vectors in MVP, swap to sqlite-vec later if needed). MLX Swift for local embedding + LLM. SpriteKit for Galaxy graph rendering. @Observable ViewModels bridge services to SwiftUI views.

**Tech Stack:** Swift/SwiftUI (macOS 26), SQLite (system libsqlite3), Accelerate (vDSP cosine similarity), MLX Swift (embedding + LLM), SpriteKit (Galaxy graph), TextKit 2 (notes editor)

**Spec:** `docs/superpowers/specs/2026-04-06-nous-mvp-design.md`

```swift
import Foundation
import SQLite3

enum DatabaseError: Error, LocalizedError {
    case openFailed(String)
    case prepareFailed(String)
    case stepFailed(String)
    case execFailed(String)

    var errorDescription: String? {
        switch self {
        case .openFailed(let msg): return "DB open failed: \(msg)"
        case .prepareFailed(let msg): return "DB prepare failed: \(msg)"
        case .stepFailed(let msg): return "DB step failed: \(msg)"
        case .execFailed(let msg): return "DB exec failed: \(msg)"
        }
    }
}

final class Database {
    private var db: OpaquePointer?

    init(path: String) throws {
        let flags = SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE | SQLITE_OPEN_FULLMUTEX
        let result = sqlite3_open_v2(path, &db, flags, nil)
        guard result == SQLITE_OK else {
            let msg = db.flatMap { String(cString: sqlite3_errmsg($0)) } ?? "unknown"
            throw DatabaseError.openFailed(msg)
        }
        try exec("PRAGMA journal_mode=WAL")
        try exec("PRAGMA foreign_keys=ON")
    }

    deinit {
        sqlite3_close(db)
    }

    func exec(_ sql: String) throws {
        var errMsg: UnsafeMutablePointer<CChar>?
        let result = sqlite3_exec(db, sql, nil, nil, &errMsg)
        if result != SQLITE_OK {
            let msg = errMsg.map { String(cString: $0) } ?? "unknown"
            sqlite3_free(errMsg)
            throw DatabaseError.execFailed(msg)
        }
    }

    func prepare(_ sql: String) throws -> Statement {
        var stmt: OpaquePointer?
        let result = sqlite3_prepare_v2(db, sql, -1, &stmt, nil)
        guard result == SQLITE_OK, let stmt else {
            let msg = db.flatMap { String(cString: sqlite3_errmsg($0)) } ?? "unknown"
            throw DatabaseError.prepareFailed(msg)
        }
        return Statement(stmt: stmt)
    }

    var lastInsertRowId: Int64 { sqlite3_last_insert_rowid(db) }
}

final class Statement {
    private let stmt: OpaquePointer

    init(stmt: OpaquePointer) {
        self.stmt = stmt
    }

    deinit {
        sqlite3_finalize(stmt)
    }

    @discardableResult
    func bind(_ index: Int32, _ value: String?) -> Statement {
        if let value {
            sqlite3_bind_text(stmt, index, (value as NSString).utf8String, -1, unsafeBitCast(-1, to: sqlite3_destructor_type.self))
        } else {
            sqlite3_bind_null(stmt, index)
        }
        return self
    }

    @discardableResult
    func bind(_ index: Int32, _ value: Int) -> Statement {
        sqlite3_bind_int64(stmt, index, Int64(value))
        return self
    }

    @discardableResult
    func bind(_ index: Int32, _ value: Double) -> Statement {
        sqlite3_bind_double(stmt, index, value)
        return self
    }

    @discardableResult
    func bind(_ index: Int32, _ value: Float) -> Statement {
        sqlite3_bind_double(stmt, index, Double(value))
        return self
    }

    @discardableResult
    func bind(_ index: Int32, _ data: Data?) -> Statement {
        if let data {
            data.withUnsafeBytes { ptr in
                sqlite3_bind_blob(stmt, index, ptr.baseAddress, Int32(data.count), unsafeBitCast(-1, to: sqlite3_destructor_type.self))
            }
        } else {
            sqlite3_bind_null(stmt, index)
        }
        return self
    }

    func step() throws -> Bool {
        let result = sqlite3_step(stmt)
        switch result {
        case SQLITE_ROW: return true
        case SQLITE_DONE: return false
        default:
            throw DatabaseError.stepFailed("step returned \(result)")
        }
    }

    func run() throws {
        _ = try step()
    }

    func columnText(_ index: Int32) -> String? {
        sqlite3_column_text(stmt, index).map { String(cString: $0) }
    }

    func columnInt(_ index: Int32) -> Int {
        Int(sqlite3_column_int64(stmt, index))
    }

    func columnDouble(_ index: Int32) -> Double {
        sqlite3_column_double(stmt, index)
    }

    func columnBlob(_ index: Int32) -> Data? {
        guard let ptr = sqlite3_column_blob(stmt, index) else { return nil }
        let size = sqlite3_column_bytes(stmt, index)
        return Data(bytes: ptr, count: Int(size))
    }

    func reset() {
        sqlite3_reset(stmt)
        sqlite3_clear_bindings(stmt)
    }
}
```

import Foundation

final class NodeStore {
    let db: Database

    init(path: String) {
        do {
            db = try Database(path: path)
            try createTables()
        } catch {
            fatalError("Failed to initialize database: \(error)")
        }
    }

    private func createTables() throws {
        try db.exec("""
            CREATE TABLE IF NOT EXISTS projects (
                id TEXT PRIMARY KEY,
                title TEXT NOT NULL,
                goal TEXT NOT NULL DEFAULT '',
                emoji TEXT NOT NULL DEFAULT '📁',
                createdAt REAL NOT NULL
            )
        """)
        try db.exec("""
            CREATE TABLE IF NOT EXISTS nodes (
                id TEXT PRIMARY KEY,
                type TEXT NOT NULL,
                title TEXT NOT NULL,
                content TEXT NOT NULL DEFAULT '',
                embedding BLOB,
                projectId TEXT REFERENCES projects(id) ON DELETE SET NULL,
                isFavorite INTEGER NOT NULL DEFAULT 0,
                createdAt REAL NOT NULL,
                updatedAt REAL NOT NULL
            )
        """)
        try db.exec("""
            CREATE TABLE IF NOT EXISTS messages (
                id TEXT PRIMARY KEY,
                nodeId TEXT NOT NULL REFERENCES nodes(id) ON DELETE CASCADE,
                role TEXT NOT NULL,
                content TEXT NOT NULL,
                timestamp REAL NOT NULL
            )
        """)
        try db.exec("""
            CREATE TABLE IF NOT EXISTS edges (
                id TEXT PRIMARY KEY,
                sourceId TEXT NOT NULL REFERENCES nodes(id) ON DELETE CASCADE,
                targetId TEXT NOT NULL REFERENCES nodes(id) ON DELETE CASCADE,
                strength REAL NOT NULL,
                type TEXT NOT NULL
            )
        """)
        try db.exec("CREATE INDEX IF NOT EXISTS idx_nodes_projectId ON nodes(projectId)")
        try db.exec("CREATE INDEX IF NOT EXISTS idx_messages_nodeId ON messages(nodeId)")
        try db.exec("CREATE INDEX IF NOT EXISTS idx_edges_sourceId ON edges(sourceId)")
        try db.exec("CREATE INDEX IF NOT EXISTS idx_edges_targetId ON edges(targetId)")
    }

    // MARK: - Nodes

    func insertNode(_ node: NousNode) throws {
        let stmt = try db.prepare("""
            INSERT INTO nodes (id, type, title, content, embedding, projectId, isFavorite, createdAt, updatedAt)
            VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8, ?9)
        """)
        stmt.bind(1, node.id.uuidString)
            .bind(2, node.type.rawValue)
            .bind(3, node.title)
            .bind(4, node.content)
            .bind(5, node.embedding.map { encodeFloats($0) })
            .bind(6, node.projectId?.uuidString)
            .bind(7, node.isFavorite ? 1 : 0)
            .bind(8, node.createdAt.timeIntervalSince1970)
            .bind(9, node.updatedAt.timeIntervalSince1970)
        try stmt.run()
    }

    func updateNode(_ node: NousNode) throws {
        let stmt = try db.prepare("""
            UPDATE nodes SET title=?1, content=?2, embedding=?3, projectId=?4,
            isFavorite=?5, updatedAt=?6 WHERE id=?7
        """)
        stmt.bind(1, node.title)
            .bind(2, node.content)
            .bind(3, node.embedding.map { encodeFloats($0) })
            .bind(4, node.projectId?.uuidString)
            .bind(5, node.isFavorite ? 1 : 0)
            .bind(6, Date().timeIntervalSince1970)
            .bind(7, node.id.uuidString)

import Foundation
import Accelerate

struct SearchResult {
    let node: NousNode
    let similarity: Float
}

final class VectorStore {
    private let nodeStore: NodeStore

    init(nodeStore: NodeStore) {
        self.nodeStore = nodeStore
    }

    /// Cosine similarity between two vectors using Accelerate vDSP.
    func cosineSimilarity(_ a: [Float], _ b: [Float]) -> Float {
        guard a.count == b.count, !a.isEmpty else { return 0 }
        var dot: Float = 0
        var normA: Float = 0
        var normB: Float = 0
        vDSP_dotpr(a, 1, b, 1, &dot, vDSP_Length(a.count))
        vDSP_dotpr(a, 1, a, 1, &normA, vDSP_Length(a.count))
        vDSP_dotpr(b, 1, b, 1, &normB, vDSP_Length(b.count))
        let denom = sqrt(normA) * sqrt(normB)
        guard denom > 0 else { return 0 }
        return dot / denom
    }

    /// Brute-force top-K search over all embedded nodes.
    func search(query: [Float], topK: Int = 5, excludeIds: Set<UUID> = []) throws -> [SearchResult] {
        let allNodes = try nodeStore.fetchNodesWithEmbeddings()
        var results: [SearchResult] = []
        for (node, embedding) in allNodes {
            guard !excludeIds.contains(node.id) else { continue }
            let sim = cosineSimilarity(query, embedding)
            results.append(SearchResult(node: node, similarity: sim))
        }
        results.sort { $0.similarity > $1.similarity }
        return Array(results.prefix(topK))
    }

    /// Find all nodes semantically similar to a given node above a threshold.
    func findSemanticNeighbors(for node: NousNode, threshold: Float = 0.75) throws -> [SearchResult] {
        guard let embedding = node.embedding else { return [] }
        let results = try search(query: embedding, topK: 50, excludeIds: [node.id])
        return results.filter { $0.similarity >= threshold }
    }

    /// Store embedding for a node.
    func storeEmbedding(_ embedding: [Float], for nodeId: UUID) throws {
        guard var node = try nodeStore.fetchNode(id: nodeId) else { return }
        node.embedding = embedding
        try nodeStore.updateNode(node)
    }
}

import Foundation
import Accelerate

struct GraphPosition {
    var x: Float
    var y: Float
}

final class GraphEngine {
    private let nodeStore: NodeStore
    private let vectorStore: VectorStore

    init(nodeStore: NodeStore, vectorStore: VectorStore) {
        self.nodeStore = nodeStore
        self.vectorStore = vectorStore
    }

    // MARK: - Force-Directed Layout

    /// Compute force-directed layout for all nodes. Returns positions keyed by node ID.
    func computeLayout(
        iterations: Int = 100,
        repulsion: Float = 5000,
        attraction: Float = 0.01,
        damping: Float = 0.9
    ) throws -> [UUID: GraphPosition] {
        let nodes = try nodeStore.fetchAllNodes()
        let edges = try nodeStore.fetchAllEdges()
        guard !nodes.isEmpty else { return [:] }

        // Initialize random positions
        var positions: [UUID: GraphPosition] = [:]
        var velocities: [UUID: GraphPosition] = [:]
        for node in nodes {
            positions[node.id] = GraphPosition(
                x: Float.random(in: -200...200),
                y: Float.random(in: -200...200)
            )
            velocities[node.id] = GraphPosition(x: 0, y: 0)
        }

        // Build adjacency for attraction
        var adjacency: [(UUID, UUID, Float)] = []
        for edge in edges {
            adjacency.append((edge.sourceId, edge.targetId, edge.strength))
        }

        for _ in 0..<iterations {
            // Repulsion between all pairs
            for i in 0..<nodes.count {
                for j in (i + 1)..<nodes.count {
                    let idA = nodes[i].id
                    let idB = nodes[j].id
                    guard var pA = positions[idA], var pB = positions[idB] else { continue }
                    let dx = pA.x - pB.x
                    let dy = pA.y - pB.y
                    let distSq = max(dx * dx + dy * dy, 1.0)
                    let force = repulsion / distSq
                    let dist = sqrt(distSq)
                    let fx = force * dx / dist
                    let fy = force * dy / dist
                    velocities[idA]!.x += fx
                    velocities[idA]!.y += fy
                    velocities[idB]!.x -= fx
                    velocities[idB]!.y -= fy
                }
            }

            // Attraction along edges
            for (srcId, tgtId, strength) in adjacency {
                guard let pA = positions[srcId], let pB = positions[tgtId] else { continue }
                let dx = pA.x - pB.x
                let dy = pA.y - pB.y
                let fx = attraction * strength * dx
                let fy = attraction * strength * dy
                velocities[srcId]!.x -= fx
                velocities[srcId]!.y -= fy
                velocities[tgtId]!.x += fx
                velocities[tgtId]!.y += fy
            }

            // Apply velocity with damping
            for node in nodes {
                velocities[node.id]!.x *= damping
                velocities[node.id]!.y *= damping
                positions[node.id]!.x += velocities[node.id]!.x
                positions[node.id]!.y += velocities[node.id]!.y
            }
        }

        return positions
    }

    // MARK: - Edge Generation

    /// Generate semantic edges for a node based on vector similarity.
    func generateSemanticEdges(for node: NousNode, threshold: Float = 0.75) throws {
        // Remove old semantic edges for this node
        try nodeStore.deleteEdges(nodeId: node.id, type: .semantic)

        let neighbors = try vectorStore.findSemanticNeighbors(for: node, threshold: threshold)
        for neighbor in neighbors {
            let edge = NodeEdge(
                sourceId: node.id,
                targetId: neighbor.node.id,
                strength: neighbor.similarity,
                type: .semantic
            )
            try nodeStore.insertEdge(edge)
        }
    }

    /// Generate shared edges for a node based on project membership.
    func generateSharedEdges(for node: NousNode) throws {
        guard let projectId = node.projectId else { return }
        try nodeStore.deleteEdges(nodeId: node.id, type: .shared)

        let siblings = try nodeStore.fetchNodes(projectId: projectId)
        for sibling in siblings where sibling.id != node.id {
            let edge = NodeEdge(
                sourceId: node.id,
                targetId: sibling.id,
                strength: 0.3, // weak shared edge
                type: .shared
            )
            try nodeStore.insertEdge(edge)
        }
    }

    /// Regenerate all edges for a node (call after embedding update).
    func regenerateEdges(for node: NousNode) throws {
        try generateSemanticEdges(for: node)
        try generateSharedEdges(for: node)
    }
}

import Foundation

@Observable
final class ChatViewModel {
    var currentNode: NousNode?
    var messages: [Message] = []
    var inputText: String = ""
    var isGenerating: Bool = false
    var currentResponse: String = ""
    var citations: [SearchResult] = []

    private let nodeStore: NodeStore
    private let vectorStore: VectorStore
    private let embeddingService: EmbeddingService
    private let graphEngine: GraphEngine
    private var llmServiceProvider: () -> (any LLMService)?

    init(
        nodeStore: NodeStore,
        vectorStore: VectorStore,
        embeddingService: EmbeddingService,
        graphEngine: GraphEngine,
        llmServiceProvider: @escaping () -> (any LLMService)?
    ) {
        self.nodeStore = nodeStore
        self.vectorStore = vectorStore
        self.embeddingService = embeddingService
        self.graphEngine = graphEngine
        self.llmServiceProvider = llmServiceProvider
    }

    func startNewConversation(title: String = "New Chat", projectId: UUID? = nil) throws {
        let node = NousNode(type: .conversation, title: title, projectId: projectId)
        try nodeStore.insertNode(node)
        currentNode = node
        messages = []
        citations = []
    }

    func loadConversation(_ node: NousNode) throws {
        currentNode = node
        messages = try nodeStore.fetchMessages(nodeId: node.id)
        citations = []
    }

    func send() async {
        let text = inputText.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !text.isEmpty, !isGenerating else { return }

        // Create conversation if needed
        if currentNode == nil {
            do {
                let title = String(text.prefix(40))
                try startNewConversation(title: title)
            } catch { return }
        }

        guard let node = currentNode else { return }

        // Save user message
        let userMsg = Message(nodeId: node.id, role: .user, content: text)
        do { try nodeStore.insertMessage(userMsg) } catch { return }
        messages.append(userMsg)
        inputText = ""
        isGenerating = true
        currentResponse = ""

        // RAG: embed query and search
        var ragCitations: [SearchResult] = []
        if embeddingService.isLoaded {
            do {
                let queryEmbedding = try await embeddingService.embed(text)
                ragCitations = try vectorStore.search(query: queryEmbedding, topK: 5, excludeIds: [node.id])
                citations = ragCitations
            } catch {
                print("RAG search failed: \(error)")
            }
        }

        // Build context
        let project: Project? = node.projectId.flatMap { try? nodeStore.fetchProject(id: $0) }
        let systemPrompt = Self.assembleContext(citations: ragCitations, projectGoal: project?.goal)

        // Build message history for LLM
        let llmMessages = messages.map { LLMMessage(role: $0.role.rawValue, content: $0.content) }

        // Generate
        guard let llm = llmServiceProvider() else {
            let fallback = Message(nodeId: node.id, role: .assistant, content: "Please configure an LLM in Settings.")
            try? nodeStore.insertMessage(fallback)
            messages.append(fallback)
            isGenerating = false
            return
        }

        do {
            let stream = try await llm.generate(messages: llmMessages, system: systemPrompt)
            for try await token in stream {
                currentResponse += token
            }
            let assistantMsg = Message(nodeId: node.id, role: .assistant, content: currentResponse)
            try nodeStore.insertMessage(assistantMsg)
            messages.append(assistantMsg)

            // Async: update embedding + edges
            Task {
                await updateNodeEmbedding(node)
            }
        } catch {
            let errorMsg = Message(nodeId: node.id, role: .assistant, content: "Error: \(error.localizedDescription)")
            try? nodeStore.insertMessage(errorMsg)
            messages.append(errorMsg)
        }

        isGenerating = false
        currentResponse = ""
    }

    /// Build system prompt with RAG context.
    static func assembleContext(citations: [SearchResult], projectGoal: String?) -> String {
        var parts: [String] = []
        parts.append("You are Nous, a personal knowledge assistant. Be concise and helpful.")

        if let goal = projectGoal, !goal.isEmpty {
            parts.append("Current project goal: \(goal)")
        }

        if !citations.isEmpty {
            parts.append("Relevant knowledge from the user's notes and conversations:")
            for (i, citation) in citations.prefix(5).enumerated() {
                let snippet = String(citation.node.content.prefix(300))
                parts.append("[\(i + 1)] \"\(citation.node.title)\" (relevance: \(Int(citation.similarity * 100))%): \(snippet)")
            }
            parts.append("Reference the above knowledge when relevant. Cite sources by title.")
        }

        return parts.joined(separator: "\n\n")
    }

    private func updateNodeEmbedding(_ node: NousNode) async {
        guard embeddingService.isLoaded else { return }
        do {
            let textToEmbed: String
            if messages.count <= 5 {
                textToEmbed = messages.map(\.content).joined(separator: "\n")
            } else {
                let recent = messages.suffix(5).map(\.content).joined(separator: "\n")
                textToEmbed = "\(node.title)\n\(recent)"
            }
            let embedding = try await embeddingService.embed(textToEmbed)
            try vectorStore.storeEmbedding(embedding, for: node.id)
            if var updated = try nodeStore.fetchNode(id: node.id) {
                updated.embedding = embedding
                try graphEngine.regenerateEdges(for: updated)
            }
        } catch {
            print("Failed to update embedding: \(error)")
        }
    }
}

final class NoteViewModel {

    var relatedNodes: [SearchResult] = []

    private let nodeStore: NodeStore
    private let vectorStore: VectorStore

    private let graphEngine: GraphEngine
    private var embedDebounceTask: Task<Void, Never>?

    init(
        nodeStore: NodeStore,
        vectorStore: VectorStore,
        embeddingService: EmbeddingService,
        graphEngine: GraphEngine
    ) {
        self.nodeStore = nodeStore
        self.vectorStore = vectorStore
        self.embeddingService = embeddingService
        self.graphEngine = graphEngine
    }

    func loadNotes() {
        do {
            notes = try nodeStore.fetchAllNodes().filter { $0.type == .note }
        } catch {
            notes = []
        }
    }

    func createNote(projectId: UUID? = nil) throws {
        let note = NousNode(type: .note, title: "Untitled", projectId: projectId)
        try nodeStore.insertNode(note)
        currentNote = note
        title = note.title
        content = ""
        loadNotes()
    }

    func openNote(_ note: NousNode) {
        currentNote = note
        title = note.title
        content = note.content
        loadRelatedNodes()
    }

    func save() {
        guard var note = currentNote else { return }
        note.title = title
        note.content = content
        do {
            try nodeStore.updateNode(note)
            currentNote = note
            loadNotes()
        } catch {
            print("Failed to save note: \(error)")
        }
        scheduleEmbedding()
    }

    func deleteNote() {
        guard let note = currentNote else { return }
        do {
            try nodeStore.deleteNode(id: note.id)
            currentNote = nil
            title = ""
            content = ""
            loadNotes()
        } catch {
            print("Failed to delete note: \(error)")
        }
    }

    func onContentChanged() {
        save()
    }

    private func scheduleEmbedding() {
        embedDebounceTask?.cancel()
        embedDebounceTask = Task {
            try? await Task.sleep(for: .seconds(2))
            guard !Task.isCancelled else { return }
            await embedCurrentNote()
        }
    }

    private func embedCurrentNote() async {
        guard let note = currentNote, embeddingService.isLoaded else { return }
        do {
            let text = "\(title)\n\(content)"
            let embedding = try await embeddingService.embed(text)
            try vectorStore.storeEmbedding(embedding, for: note.id)
            if var updated = try nodeStore.fetchNode(id: note.id) {
                updated.embedding = embedding
                try graphEngine.regenerateEdges(for: updated)
            }
            loadRelatedNodes()
        } catch {
            print("Failed to embed note: \(error)")
        }
    }

    private func loadRelatedNodes() {
        guard let note = currentNote, let embedding = note.embedding else {
            relatedNodes = []
            return
        }
        do {
            relatedNodes = try vectorStore.search(query: embedding, topK: 5, excludeIds: [note.id])
        } catch {
            relatedNodes = []
        }
    }
}

        isLoading = false
    }

        positions[nodeId] = GraphPosition(x: x, y: y)
    }

    func nodeForId(_ id: UUID) -> NousNode? {
        nodes.first { $0.id == id }

import SpriteKit
import Foundation

final class GalaxyScene: SKScene {
    var graphNodes: [NousNode] = []
    var graphEdges: [NodeEdge] = []
    var positions: [UUID: GraphPosition] = [:]
    var onNodeTapped: ((UUID) -> Void)?

    private var nodeSprites: [UUID: SKNode] = [:]
    private var edgeShapes: [SKShapeNode] = []
    private var draggedNode: UUID?

    override func didMove(to view: SKView) {
        backgroundColor = NSColor(red: 26/255, green: 26/255, blue: 46/255, alpha: 1)
        anchorPoint = CGPoint(x: 0.5, y: 0.5)

        // Add camera for zoom/pan
        let cam = SKCameraNode()
        camera = cam
        addChild(cam)

        rebuildScene()
    }

    func rebuildScene() {
        removeAllChildren()
        nodeSprites.removeAll()
        edgeShapes.removeAll()

        // Draw edges
        for edge in graphEdges {
            guard let p1 = positions[edge.sourceId], let p2 = positions[edge.targetId] else { continue }
            let path = CGMutablePath()
            path.move(to: CGPoint(x: CGFloat(p1.x), y: CGFloat(p1.y)))
            path.addLine(to: CGPoint(x: CGFloat(p2.x), y: CGFloat(p2.y)))
            let line = SKShapeNode(path: path)
            line.strokeColor = NSColor(red: 243/255, green: 131/255, blue: 53/255, alpha: CGFloat(edge.strength * 0.4))
            line.lineWidth = CGFloat(1 + edge.strength)
            addChild(line)
            edgeShapes.append(line)
        }

        // Draw nodes
        for node in graphNodes {
            guard let pos = positions[node.id] else { continue }
            let contentLen = Float(node.content.count)
            let radius = CGFloat(max(12, min(30, 12 + sqrt(contentLen) * 0.5)))

            let circle = SKShapeNode(circleOfRadius: radius)
            circle.position = CGPoint(x: CGFloat(pos.x), y: CGFloat(pos.y))
            circle.fillColor = NSColor(red: 243/255, green: 131/255, blue: 53/255, alpha: 0.7)
            circle.strokeColor = NSColor(red: 243/255, green: 131/255, blue: 53/255, alpha: 0.3)
            circle.lineWidth = 2
            circle.glowWidth = 4
            circle.name = node.id.uuidString

            // Emoji label
            let emojiLabel = SKLabelNode(text: node.type == .conversation ? "💬" : "📝")
            emojiLabel.fontSize = radius * 0.8
            emojiLabel.verticalAlignmentMode = .center
            circle.addChild(emojiLabel)

            // Title label below
            let titleLabel = SKLabelNode(text: node.title)
            titleLabel.fontSize = 10
            titleLabel.fontColor = NSColor(white: 1, alpha: 0.7)
            titleLabel.position = CGPoint(x: 0, y: -radius - 14)
            titleLabel.horizontalAlignmentMode = .center
            circle.addChild(titleLabel)

            addChild(circle)
            nodeSprites[node.id] = circle
        }
    }

    override func mouseDown(with event: NSEvent) {
        let location = event.location(in: self)
        let tappedNodes = nodes(at: location)
        for tapped in tappedNodes {
            if let name = tapped.name, let uuid = UUID(uuidString: name) {
                draggedNode = uuid
                return
            }
            if let parentName = tapped.parent?.name, let uuid = UUID(uuidString: parentName) {
                draggedNode = uuid
                return
            }
        }
    }

    override func mouseDragged(with event: NSEvent) {
        guard let nodeId = draggedNode, let sprite = nodeSprites[nodeId] else { return }
        let location = event.location(in: self)
        sprite.position = location
        positions[nodeId] = GraphPosition(x: Float(location.x), y: Float(location.y))
    }

    override func mouseUp(with event: NSEvent) {
        if let nodeId = draggedNode {
            let location = event.location(in: self)
            let startPos = nodeSprites[nodeId]?.position ?? .zero
            let distance = sqrt(pow(location.x - startPos.x, 2) + pow(location.y - startPos.y, 2))
            // If barely moved, treat as tap
            if distance < 5 {
                onNodeTapped?(nodeId)
            }
        }
        draggedNode = nil
    }

    override func scrollWheel(with event: NSEvent) {
        // Zoom the entire scene by adjusting camera scale
        guard let camera else { return }
        let newScale = camera.xScale * (1 - event.deltaY * 0.02)
        let clamped = max(0.3, min(3.0, newScale))
        camera.setScale(clamped)
    }

    override func magnify(with event: NSEvent) {
        // Pinch-to-zoom via trackpad
        guard let camera else { return }
        let newScale = camera.xScale * (1 - event.magnification)
        let clamped = max(0.3, min(3.0, newScale))
        camera.setScale(clamped)
    }
}
